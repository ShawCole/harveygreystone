```js
  // --- Pipeline stages (v2) ---
  // `stage` is the primary pipeline state; legacy `status` is derived from it
  // via stageToStatus() so filters/kanban/charts/exports keep working.
  const STAGES = [
    { id: 'intake',           label: 'Link Sent' },
    { id: 'docs_in',          label: 'Docs Coming In' },
    { id: 'intake_complete',  label: 'Intake Complete' },
    { id: 'ai_vetted',        label: 'AI Vetted' },
    { id: 'revisit_client',   label: 'Revisit with Client (call)' },
    { id: 'good_deal',        label: 'Good Deal' },
    { id: 'summary_review',   label: 'Project Summary Review' },
    { id: 'contract',         label: 'Client Contract / Success Fee' },
    { id: 'underwriting',     label: 'Official Underwriting' },
    { id: 'letterhead',       label: 'Letterhead / Term Sheet' },
    { id: 'signing',          label: 'Signing' },
    { id: 'funding_date',     label: 'Funding Date Set' },
    { id: 'funded',           label: 'Funded' },
  ];
  const STAGE_LABEL = Object.fromEntries(STAGES.map(s => [s.id, s.label]));
  STAGE_LABEL.declined = 'Declined';

  function stageToStatus(stage) {
    const i = STAGES.findIndex(s => s.id === stage);
    if (stage === 'funded') return 'closed';
    if (i >= 8) return 'negotiation';   // underwriting..funding_date
    if (i >= 4) return 'proposal';      // revisit_client..contract
    return 'lead';                      // intake..ai_vetted (and declined/unknown)
  }
  function statusToStage(status) {
    return { lead: 'intake', proposal: 'good_deal', negotiation: 'underwriting', closed: 'funded' }[status] || 'intake';
  }

  function recordHistory(deal, note, stage) {
    deal.history = deal.history || [];
    deal.history.push({ at: new Date().toISOString(), stage: stage || deal.stage || '', note, actor: 'team' });
  }

  function setStage(dealId, stageId) {
    const d = deals.find(x => x.id === dealId);
    if (!d || d.stage === stageId) return;
    d.stage = stageId;
    d.status = stageId === 'declined' ? d.status : stageToStatus(stageId);
    recordHistory(d, `Moved to ${STAGE_LABEL[stageId] || stageId}`, stageId);
    saveData();
    renderDealPane(dealId);      // defined in Task 3
    renderDealRail();            // defined in Task 2
  }
```
